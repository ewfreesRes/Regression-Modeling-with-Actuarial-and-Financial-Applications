```yaml
type: TwoColumns
key: ed77bae84e
```

`@part1`
```
(mparent <- mean(ht_par))
(sdparent <- sd(ht_par))
x <- seq(60, 80,by = 0.1)
hist(ht_par, freq = FALSE)
lines(x, dnorm(x, mean = mparent, 
          sd = sdparent), col = "blue")
```

```
   sd = sdparent)

```

`@part2`
```
pnorm(72, mean = mean(ht_par), 
   sd = sd(ht_par))
> [1] 0.9238302
```

`@part3`
```
(StdUnitsTallHeight <- 
   (TallHeight - mparent)/sdparent)
> [1] 1.431317
```

`@part4`
```
pnorm(StdUnitsTallHeight, mean = 0, 
   sd = 1)
> [1] 0.9238302
```
